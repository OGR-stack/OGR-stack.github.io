Method 1: Token Validation

This method validates the authenticity of a Discord token.

function validateToken(token) {
    const regex = /^[\w-]{24}\.[\w-]{6}\.[\w-]{27}$/;
    return regex.test(token);
}

Method 2: Cooldown Check

Check if a Discord boost token is on cooldown.

async function checkCooldown(token) {
    const response = await fetch('https://discord.com/api/v9/users/@me', {
        headers: { 'Authorization': `Bearer ${token}` }
    });
    const data = await response.json();
    return data.premium_since ? 'On cooldown' : 'Available';
}
